# password_generator.py
# -----------------------
# ランダムな安全パスワードを生成
# -----------------------

import random
import string

length = int(input("パスワードの長さを入力: "))

chars = string.ascii_letters + string.digits + string.punctuation
password = ''.join(random.choice(chars) for _ in range(length))

print("生成されたパスワード:", password)
